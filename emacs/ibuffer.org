    "Setup ibuffer defaults."
    (require 'ibuf-ext)
    (ibuffer-auto-mode 1)
    (ibuffer-switch-to-saved-filter-groups "home")
    (toggle-truncate-lines +1))
  (add-hook 'ibuffer-mode-hook #'actuator-ibuffer-setup)
